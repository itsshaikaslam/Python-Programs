#Write a program to accept a number from user and print its factorial
def getFactorial(num):
	fact=-1
	if num > 0:
		if num < 3:
			fact=num
		else:
			fact=1
			while num!=1:
				fact=fact*num
				num-=1
	return fact
def main():
	num=input("Enter number :")
	print("Factorial of {} is {}".format(num,getFactorial(num)))
if __name__=="__main__":
	main()
